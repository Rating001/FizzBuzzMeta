                            buzzNumber = parseInt(buzzNumber);
                        
                            if (Number.isInteger(fizzNumber) && Number.isInteger(buzzNumber)) {

                                    switch(choice){
                                        case 'ternaryWithTable':
                                            answers = checkDivisibleA(fizzNumber, buzzNumber);
                                            displayTable(answers,checked);
                                            break;
                                        case 'ternaryWithText':
                                            answers = checkDivisibleB(fizzNumber, buzzNumber);
                                            if (checked) {
                                                coloredAnswers = addColorText(answers);
                                                displayText(coloredAnswers);
                                            } else {
                                                displayText(answers);
                                            }
                                            break;
                                        case 'switchWithText':
                                            answers = checkDivisibleC(fizzNumber, buzzNumber);
                                            if (checked) {
                                                coloredAnswers = addColorText(answers);
                                                displayText(coloredAnswers);
                                            } else {
                                                displayText(answers);
                                            }
                                            break;
                                        case 'ifStatementWithText':
                                            answers = checkDivisibleD(fizzNumber, buzzNumber);
                                            if (checked) {
                                                coloredAnswers = addColorText(answers);
                                                displayText(coloredAnswers);
                                            } else {
                                                displayText(answers);
                                            }
                                            break;
                                    }
                        
                                } else {
                                    alert("You must enter numbers greater than 0.");
                                }
                            } else {
                                alert("You must enter integers.");
                            }
                            }

                        //This solution is specifically for the "Ternary Operator with Table Output" option
                        function checkDivisibleA(fizz,buzz) {
                            let answers = [];
                        
                            for (let i = 1; i <= 100; i++) {
                                answers[i] = ((i % fizz == 0 ? 'Fizz' : '') + (i % buzz == 0 ? 'Buzz' : '') || i);        
                            }
                        
                            return answers;
                        }

                        //Function to display the elements in a table
                        function displayTable(answers,checked) {
                        
                            let tableBody = document.getElementById('resultsTable');
                        
                            let templateRow = document.getElementById('fbTemplate');
                        
                            tableBody.innerHTML = "";
                        
                            for (let i = 1; i < 100; i += 5) {
                                let tableRow = document.importNode(templateRow.content, true);
                                
                                //Place the TDs into an array and write values to them
                                //If the color switch is on add the classes for 'Fizz', 'Buzz' and 'FizzBuzz' using the array so they can be manipulated via CSS
                                let rowCols = tableRow.querySelectorAll("td");
                        
                                if (checked) {
                                    rowCols[0].classList.add(answers[i]);
                                    rowCols[0].textContent = answers[i];
                                } else {
                                    rowCols[0].textContent = answers[i];
                                }
                                if (checked) {
                                    rowCols[1].classList.add(answers[i+1]);
                                    rowCols[1].textContent = answers[i+1];
                                } else {
                                    rowCols[1].textContent = answers[i+1];
                                }
                                if (checked) {
                                    rowCols[2].classList.add(answers[i+2]);
                                    rowCols[2].textContent = answers[i+2];
                                } else {
                                    rowCols[2].textContent = answers[i+2];
                                }
                                if (checked) {
                                    rowCols[3].classList.add(answers[i+3]);
                                    rowCols[3].textContent = answers[i+3];
                                } else {
                                    rowCols[3].textContent = answers[i+3];
                                }
                                if (checked) {
                                    rowCols[4].classList.add(answers[i+4]);
                                    rowCols[4].textContent = answers[i+4];
                                } else {
                                    rowCols[4].textContent = answers[i+4];
                                }
                        
                                //Print to the table
                                tableBody.appendChild(tableRow);
                            }
                        
                            //Turn on the alert box
                            document.getElementById("alertTable").classList.remove("hideMe");
                        
                        }

                        function checkDivisibleB(fizz,buzz) {
                            let answers = "";
                        
                            for (let index = 1; index <= 100; index++){

                        //--------------------------------------------------------------------------------
                        //Add color to the string of answers if the switch is enabled
                        function addColorText(answers) {
                        
                            coloredAnswers = "";
                            let splitString = answers.split(" ");
                        
                            for (i = 0 ; i <= 99; i++) {
                                switch (splitString[i]) {
                                    case 'FizzBuzz': {
                                        coloredAnswers += `FizzBuzz `;
                                        break;
                                    }
                                    case 'Fizz': {
                                        coloredAnswers += `Fizz `;
                                        break;
                                    }
                                    case 'Buzz': {
                                        coloredAnswers += `Buzz `;
                                        break;
                                    }
                                    default: {
                                        coloredAnswers += `${splitString[i]} `;
                                        break;
                                    }
                                }
                            }
                        
                            return coloredAnswers;
                        }

                        //--------------------------------------------------------------------------------
                        //This solution is specifically for the "Boolean Switch Case with Text Output" option
                        function checkDivisibleC(fizz, buzz) {
                            let answers = "";
                            let fizzy = false;
                            let buzzy = false;
                        
                            for (let i = 1; i <= 100; i++) {
                                fizzy = (i % fizz == 0);
                                buzzy = (i % buzz == 0);
                                switch(true) {
                                    case fizzy && buzzy: {
                                        answers += 'FizzBuzz ';
                                        break;
                                    }
                                    case fizzy: {
                                        answers += 'Fizz ';
                                        break;
                                    }
                                    case buzzy: {
                                        answers += 'Buzz ';
                                        break;
                                    }
                                    default: {
                                        answers += (i + ' ');
                                        break;
                                    }
                                }
                            }
                        
                            return answers;
                        }

Ternary with Table Output

The Code is constructed in three functions.

getInput()
  1. getInput() is called by an event listener.
  2. Sets the given Fizz and Buzz values to variables.
  3. Checks if the color switch is activated and sets a variable to true/false.
  4. Sets the user's solution "choice" to a variable.
  5. Parses Fizz and Buzz into Integers.
  6. Checks that Fizz and Buzz are both numbers.
  7. Calls the specific function based on the given options with a switch statement.
checkDivisibleA(fizz,buzz)
  1. Clears the "answers" variable.
  2. Uses a For loop to run 1 to 100.
  3. Uses the Ternary Operator to write FizzBuzz, Fizz, Buzz, or the loop number to the "answers" array.
  4. Returns "answers".
displayTable(answers,checked)
  1. Sets the HTML element "resultsTable" to a variable, then clears it.
  2. Sets an HTML Template to a variable.
  3. Uses a For loop to append the values in "answers[]" into the template 5 rows at a time.
      4. -If the "Add Color" switch is enabled it adds the classes "Fizz", "Buzz", or "FizzBuzz" so they can be manipulated using CSS.
  4. Unhides the table on the page.
